    <script>
        function loadDoc() {
          const xhttp = new XMLHttpRequest();
          xhttp.onload = function() {
            const response = JSON.parse(this.responseText);
            console.log(response)

            response.forEach(item => {
                document.getElementById('table_data').innerHTML += `<tr class="border border-slate-700 bg-slate-500"><td>${item.id}</td> <td>${item.nama}</td> <td>${item.nim}</td> <td>${item.jurusan}</td>`;
            });
          }
          xhttp.open("GET", "https://lucis.serveo.net/belajar-php/mahasiswa/mahasiswa.php");
          xhttp.send();
        }
    </script>
